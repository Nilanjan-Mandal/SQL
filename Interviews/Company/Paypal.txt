Answer: 

SELECT department_id from 
            (SELECT e.*, 
            AVG(salary) OVER(PARTITION BY department_id) as avg_ep
            from emp_paypal as e) as x
            where x.avg_ep < (SELECT AVG(salary) as AVG from emp_paypal)
